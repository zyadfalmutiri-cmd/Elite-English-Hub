<script>

let currentUser = null;

function getUsers() {
  return JSON.parse(localStorage.getItem("users") || "{}");
}

function saveUsers(users) {
  localStorage.setItem("users", JSON.stringify(users));
}

/* NOTIFY */
function notify(msg) {
  let n = document.createElement("div");
  n.innerText = msg;
  n.style = "position:fixed;bottom:80px;left:50%;transform:translateX(-50%);background:#38bdf8;color:black;padding:10px;border-radius:10px;";
  document.body.appendChild(n);
  setTimeout(() => n.remove(), 2000);
}

/* REGISTER */
function register() {
  let users = getUsers();

  let u = document.getElementById("username").value.trim();
  let p = document.getElementById("password").value.trim();

  if (!u || !p) return notify("Fill all fields");

  if (users[u]) return notify("User exists");

  users[u] = {
    password: p,
    xp: 0,
    streak: 0,
    lastActive: null
  };

  saveUsers(users);
  notify("Account created");
}

/* LOGIN */
function login() {
  let users = getUsers();

  let u = document.getElementById("username").value.trim();
  let p = document.getElementById("password").value.trim();

  if (!users[u] || users[u].password !== p)
    return notify("Invalid login");

  currentUser = u;

  document.getElementById("auth").style.display = "none";
  document.getElementById("app").style.display = "block";

  updateUI();
}

/* LOGOUT */
function logout() {
  currentUser = null;
  location.reload();
}

/* LESSON COMPLETE */
function completeLesson(xp) {
  let users = getUsers();
  let u = users[currentUser];

  u.xp += xp;

  let today = new Date().toDateString();

  if (u.lastActive !== today) {
    u.streak += 1;
    u.lastActive = today;
  }

  saveUsers(users);
  updateUI();

  notify("+" + xp + " XP");
}

/* LEVEL */
function getLevel(xp) {
  return Math.floor(xp / 100) + 1;
}

/* UPDATE UI */
function updateUI() {
  let users = getUsers();
  let u = users[currentUser];

  document.getElementById("user").innerText = currentUser;
  document.getElementById("xp").innerText = u.xp;
  document.getElementById("level").innerText = getLevel(u.xp);
  document.getElementById("streak").innerText = u.streak;

  let percent = (u.xp % 100);
  document.getElementById("bar").style.width = percent + "%";
}

/* NAV */
function showPage(page) {
  document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));
  document.getElementById(page).classList.add("active");
}
